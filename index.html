    <script>

        const players = [];

        const categories = [

            { id: 'ones', label: 'Ykköset' },

            { id: 'twos', label: 'Kakkoset' },

            { id: 'threes', label: 'Kolmoset' },

            { id: 'fours', label: 'Neloset' },

            { id: 'fives', label: 'Viitoset' },

            { id: 'sixes', label: 'Kuutoset' },

            { id: 'sum', label: 'Välisumma', readonly: true, class: 'bold-row' },

            { id: 'bonus', label: 'Bonus (50)', readonly: true, class: 'bold-row' },

            { id: 'pair', label: 'Pari' },

            { id: 'two-pair', label: 'Kaksi paria' },

            { id: 'three-kind', label: 'Kolmosluku' },

            { id: 'four-kind', label: 'Nelosluku' },

            { id: 'small-str', label: 'Pieni suora' },

            { id: 'large-str', label: 'Suuri suora' },

            { id: 'full-house', label: 'Täyskäsi' },

            { id: 'chance', label: 'Sattuma' },

            { id: 'yatzy', label: 'Yatzy' },

            { id: 'total', label: 'Yhteensä', readonly: true, class: 'total-row' }

        ];


        function addPlayer() {

            const input = document.getElementById('player-name');

            const name = input.value.trim();

            

            if (name && players.length < 5) {

                players.push(name);

                input.value = '';

                document.getElementById('player-list').innerText = "Pelaajat: " + players.join(', ');

                document.getElementById('start-btn').disabled = false;

                

                if (players.length === 5) {

                    document.getElementById('add-btn').disabled = true;

                    input.disabled = true;

                }

            }

        }


        function startGame() {

            document.getElementById('setup').style.display = 'none';

            document.getElementById('game-container').style.display = 'block';

            

            const headerRow = document.getElementById('header-row');

            const tbody = document.getElementById('table-body');


            // Lisää pelaajien nimet otsikoiksi

            players.forEach(name => {

                const th = document.createElement('th');

                th.innerText = name;

                headerRow.appendChild(th);

            });


            // Luo rivit

            categories.forEach(cat => {

                const tr = document.createElement('tr');

                if (cat.class) tr.className = cat.class;


                const labelTd = document.createElement('td');

                labelTd.innerText = cat.label;

                labelTd.className = 'row-label sticky-col';

                tr.appendChild(labelTd);


                players.forEach((_, pIdx) => {

                    const td = document.createElement('td');

                    if (cat.readonly) {

                        td.id = `score-${cat.id}-${pIdx}`;

                        td.innerText = '0';

                        td.className = 'sum-display';

                    } else {

                        const input = document.createElement('input');

                        input.type = 'number';

                        input.className = 'score-input';

                        input.dataset.player = pIdx;

                        input.dataset.cat = cat.id;

                        input.oninput = () => calculateScores(pIdx);

                        td.appendChild(input);

                    }

                    tr.appendChild(td);

                });

                tbody.appendChild(tr);

            });

        }


        function calculateScores(pIdx) {

            const inputs = document.querySelectorAll(`.score-input[data-player="${pIdx}"]`);

            let upperSum = 0;

            let totalSum = 0;


            const upperCats = ['ones', 'twos', 'threes', 'fours', 'fives', 'sixes'];


            inputs.forEach(input => {

                const val = parseInt(input.value) || 0;

                const catId = input.dataset.cat;


                if (upperCats.includes(catId)) {

                    upperSum += val;

                }

                totalSum += val;

            });


            // Välisumma ja Bonus

            const bonus = upperSum >= 63 ? 50 : 0;

            document.getElementById(`score-sum-${pIdx}`).innerText = upperSum;

            document.getElementById(`score-bonus-${pIdx}`).innerText = bonus;


            // Lopullinen summa (sisältää bonuksen)

            document.getElementById(`score-total-${pIdx}`).innerText = totalSum + bonus;

        }
